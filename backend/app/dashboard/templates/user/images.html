API Reference (click to expand)

Generate images programmatically using the OpenAI-compatible API:

cURL
curl -X POST https://mindrouter.uidaho.edu/v1/images/generations \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -d '{
    "prompt": "A serene mountain landscape at sunset",
    "model": "{{ default_model }}",
    "size": "{{ default_size }}",
    "num_inference_steps": {{ default_steps }},
    "guidance_scale": {{ default_guidance }},
    "response_format": "b64_json"
  }'
Python
from openai import OpenAI

client = OpenAI(
    base_url="https://mindrouter.uidaho.edu/v1",
    api_key="YOUR_API_KEY",
)

response = client.images.generate(
    model="{{ default_model }}",
    prompt="A serene mountain landscape at sunset",
    size="{{ default_size }}",
    response_format="b64_json",
)
image_data = response.data[0].b64_json
JavaScript
const response = await fetch('https://mindrouter.uidaho.edu/v1/images/generations', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'Authorization': 'Bearer YOUR_API_KEY',
  },
  body: JSON.stringify({
    prompt: 'A serene mountain landscape at sunset',
    model: '{{ default_model }}',
    size: '{{ default_size }}',
    response_format: 'b64_json',
  }),
});
const data = await response.json();
const imageBase64 = data.data[0].b64_json;
